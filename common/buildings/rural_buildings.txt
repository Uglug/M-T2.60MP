mines_1 = {
	cost = 100
	time = 12
	
	trigger = {
		OR = {
			can_build_mines_1 = yes
			AND = {
				can_keep_mines_1 = yes
				has_building = mines_1
			}
		}
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = { hidden_effect = { recalc_infra_rural = yes } }
	on_destroyed = { hidden_effect = { recalc_infra_rural = yes } }
	on_obsolete = {}
}
